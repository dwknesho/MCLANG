ORDER_START

!* Void procedures (EMPTY), String operations, Escape sequences, Nested loops,
Array expression indexing, Compound assignment operators, Modulo operations,
Unary operators, Decrement,  Loop control (continue), Comparison operator 
(!=), Exception handling with finally block *!

    !! Void procedure - no YIELD
    TASK EMPTY printBanner(RECIPE msg) {
        SERVE "\n========================================\n  [ ";
        SERVE msg;
        SERVE " ]\n========================================\n";
    }

    PRICE scores[6];
    PRICE total = 100;
    PRICE result;
    RECIPE firstName = "John";
    RECIPE lastName = "Doe";
    RECIPE fullName;

    printBanner("Features Demo");

    !! String concatenation with variables
    fullName = firstName + " " + lastName;
    SERVE "> Student:\t";
    SERVE fullName;

    !! All escape sequences: \", \\, \n, \t
    SERVE "\n> Path test:\t\"C:\\Users\\Data\"\n\t\tEnd of path\n";

    !! Nested loops with expression as array index
    SERVE "\n> Filling scores array...\n";
    PREP (PRICE i = 0; i < 2; i++) {
        SERVE "\t";
        PREP (PRICE j = 0; j < 3; j++) {
            scores[i * 3 + j] = (i + 1) * 10 + j;  !! Expression as index
            
            SERVE "[";
            SERVE i * 3 + j;
            SERVE "]:";
            SERVE scores[i * 3 + j];
            SERVE "   ";
        }
        SERVE "\n";
    }

    !! Compound assignment operators
    total *= 3;   !! 100 * 3 = 300
    total /= 2;   !! 300 / 2 = 150
    total %= 40;  !! 150 % 40 = 30

    SERVE "\n> After operations (total): ";
    SERVE total;

    !! Unary operators
    result = +total;  !! Unary plus
    result = -total;  !! Unary minus (result = -30)
    SERVE "\n> Unary minus result:       ";
    SERVE result;
    SERVE "\n";

    !! Decrement and NEXT (continue)
    SERVE "\n> Odd numbers only: \n\t";
    PRICE count = 10;
    REFILL (count > 0) {
        count--;  !! Decrement
        
        WANT (count % 2 == 0) {  !! Modulo to check even
            NEXT;  !! Skip even numbers
        }
        
        SERVE count;
        SERVE "  "; !! Spaces out the numbers
        
        WANT (count <= 3) {
            DONE;
        }
    }
    SERVE "\n";

    !! Inequality operator
    WANT (count != 0) {
        SERVE "\n> Inequality check passed: Count is not zero.\n";
    }

    !! Exception with CLEAN (finally)
    SERVE "\n> Testing exception...\n";
    CHECK {
        WANT (total < 50) {
            SPILL "Value too low";
        }
        SERVE "   - Success!\n";
    } HANDLE (err) {
        SERVE "   - Error caught: ";
        SERVE err;
        SERVE "\n";
    } CLEAN {
        SERVE "   - Cleanup always runs (Finally block executed)\n";
    }

    printBanner("Execution Complete");

ORDER_END
