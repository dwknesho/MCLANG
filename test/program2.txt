ORDER_START

!* declaration, i/o, switch, loops, exception handling, logical operators, 
string concatenation, increment/decrement, and boolean expressions *!

    PRICE itemCode;
    PRICE quantity;
    PRICE unitPrice;
    PRICE subtotal = 0;
    PRICE total = 0;
    PRICE itemCount = 0;
    QUALITY running = FRESH;
    QUALITY hasDiscount = EXPIRED;

    SERVE "\n=== McLang Shopping Cart ===\n\n";

    !! keep adding items until user enters 0
    REFILL (running == FRESH) {

        SERVE "Enter item code (1=Burger, 2=Fries, 3=Drink, 0=Checkout): ";
        ORDER itemCode;

        WANT (itemCode == 0) {
            running = EXPIRED;
        } ONLY {
            !! switch to get unit price
            MENU (itemCode) {
                FOOD 1 :
                    unitPrice = 120;
                    SERVE " > Burger selected.\n";
                    DONE;
                FOOD 2 :
                    unitPrice = 60;
                    SERVE " > Fries selected.\n";
                    DONE;
                FOOD 3 :
                    unitPrice = 45;
                    SERVE " > Drink selected.\n";
                    DONE;
                SOLDOUT :
                    SERVE " > Invalid item code. Please try again.\n\n";
                    unitPrice = 0;
            }

            !! only process valid items
            WANT (unitPrice > 0) {
                SERVE "Enter quantity: ";
                ORDER quantity;

                CHECK {
                    WANT (quantity <= 0) {
                        SPILL "Quantity must be greater than zero";
                    }
                    subtotal = unitPrice * quantity;
                    total += subtotal;
                    itemCount++;
                    
                    SERVE " > Subtotal for this item: ";
                    SERVE subtotal;
                    SERVE "\n\n"; !! Adds a clean break before the menu loops again
                    
                } HANDLE (err) {
                    SERVE " > Invalid quantity entered. Item skipped.\n\n";
                }
            }
        }
    }

    !! apply discount if total exceeds 500 and has 3 or more items
    hasDiscount = total >= 500 && itemCount >= 3;
    WANT (hasDiscount == FRESH) {
        total -= 50;
        SERVE "\n*** Discount of 50 applied! ***\n";
    }

    !! Print formatted receipt
    SERVE "\n========================\n";
    SERVE "     Order Summary      \n";
    SERVE "========================\n";
    SERVE "Items ordered: ";
    SERVE itemCount;
    SERVE "\nTotal amount:  ";
    SERVE total;
    SERVE "\n========================\n\n";

ORDER_END
